#define CHECK_CUDA_ERROR(call) { \       // Macro definition starts
    cudaError_t err = call; \           // Execute CUDA call and store result
    if (err != cudaSuccess) { \         // Check if error occurred
        printf("CUDA Error: %s at line %d\n", \
               cudaGetErrorString(err), \  // Convert error code to string
               __LINE__); \                // Built-in macro for current line number
        exit(1); \                        // Exit program if error
    } \
}
